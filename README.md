# How To Use
- ### Lite UI
```lua
local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/SxnwDev/UI/main/LiteUI.lua"))()
local window = library:new()

local page = window:addPage()
local section = page:addSection({ Divisions = 1 })
section:addButton()
section:addClipboardLabel()
section:addDualLabel()
section:addLabel({ text = '<font color="#' .. library.Settings.theme.Accent:ToHex() .. '"><b>Lite</b></font>, the best UI ^-^' })
section:addSlider({ Max = 10, Min = 0, Default = 0 })
section:addToggle()
section:addCheckbox({ Group = "group_name" })
section:addCheckbox({ Group = "group_name" })
section:addToggle({ Title = "Toggle + KeyBind", KeyBind = true })
section:addKeybind()
section:addDropdown({ List = { 1, 2, 3, 4, 5 } })
section:addDropdown({ Title = "Multi Dropdown", List = { 1, 2, 3, 4, 5 }, Multi = true })

local page = window:addPage()
local section = page:addSection({ Divisions = 2 })
section:addButton({ Section = 2 })
section:addClipboardLabel({ Section = 2 })
section:addDualLabel({ Section = 2 })
section:addLabel({ text = '<font color="#' .. library.Settings.theme.Accent:ToHex() .. '"><b>Lite</b></font>, the best UI ^-^', Section = 2 })
section:addSlider({ Max = 10, Min = 0, Default = 0, Section = 2 })
section:addToggle({ Section = 2 })
section:addCheckbox({ Section = 2, Group = "group_name_2" })
section:addCheckbox({ Section = 2, Group = "group_name_2" })
section:addToggle({ Title = "Toggle + KeyBind", KeyBind = true, Section = 2 })
section:addKeybind({ Section = 2 })
section:addDropdown({ Section = 2, List = { 1, 2, 3, 4, 5 } })
section:addDropdown({ Section = 2, Title = "Multi Dropdown", List = { 1, 2, 3, 4, 5 }, Multi = true })

local page = window:addPage()
```
